\definecolor{sectionblue}{RGB}{58,96,177}

% ---------- Section title formatting ----------
% Bold, all-caps, with a rule underneath
\titleformat{\section}
  {\large\bfseries\color{sectionblue}}
  {}{0em}{\MakeUppercase}
  [\vspace{-14pt}\rule{\linewidth}{0.8pt}\vspace{-4pt}]

\titlespacing*{\section}{0pt}{10pt}{6pt}

% ---------- Section body padding ----------
% Adds horizontal padding between section titles and their body content.
\newlength{\sectionbodypadding}
\setlength{\sectionbodypadding}{1.0em}
\newif\ifcvsectionbodyopen
\cvsectionbodyopenfalse
\let\cvorigsection\section
\renewcommand{\section}[1]{%
  \ifcvsectionbodyopen
    \end{adjustwidth}%
    \cvsectionbodyopenfalse
  \fi
  \cvorigsection{#1}%
  \begin{adjustwidth}{\sectionbodypadding}{\sectionbodypadding}%
  \cvsectionbodyopentrue
}
\AtEndDocument{%
  \ifcvsectionbodyopen
    \end{adjustwidth}%
  \fi
}

% ---------- List settings ----------

  itemsep    = 1pt,
  parsep     = 0pt
}

% ---------- No paragraph indent ----------
\setlength{\parindent}{0pt}

% ---------- Helper: two-column dated entry ----------
% Usage: \entry{LEFT TEXT}{RIGHT (year/date)}
\newcommand{\entry}[2]{%
  \noindent
  \begin{tabularx}{\linewidth}{@{}X r@{}}
    #1 & #2
  \end{tabularx}\vspace{2pt}
}

% ---------- Helper: award/honor entry with bold title ----------
% Usage: \award{Year}{Award Name}{Venue/Notes}
\newcommand{\award}[3]{%
  \entry{#1 \quad #2 \textit{#3}}{}%
}
