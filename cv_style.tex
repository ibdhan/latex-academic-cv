% ---------- Section title formatting ----------
% Bold, all-caps, with a rule underneath
\titleformat{\section}
  {\large\bfseries\uppercase}
  {}{0em}{}
  [\vspace{-4pt}\rule{\linewidth}{0.8pt}\vspace{-4pt}]

\titlespacing*{\section}{0pt}{10pt}{6pt}

% ---------- List settings ----------

  leftmargin = 1.5em,

  itemsep    = 1pt,
  parsep     = 0pt
}

% ---------- No paragraph indent ----------
\setlength{\parindent}{0pt}
\setlength{\parskip}{4pt}

% ---------- Helper: two-column dated entry ----------
% Usage: \entry{LEFT TEXT}{RIGHT (year/date)}
\newcommand{\entry}[2]{%
  \noindent
  \begin{tabularx}{\linewidth}{@{}X r@{}}
    #1 & #2
  \end{tabularx}\vspace{2pt}
}

% ---------- Helper: award/honor entry with bold title ----------
% Usage: \award{Year}{Award Name}{Venue/Notes}
\newcommand{\award}[3]{%
  \entry{#1 \quad #2 \textit{#3}}{}%
}
